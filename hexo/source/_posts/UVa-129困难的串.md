const int N = 100;
int n, l, a[N];

void dfs(int &k, int d) {
	if (k == n) {
		F(i, d) {
			if (i == 64) printf("\n");
			else if (i && (i % 4) == 0) printf(" ");
			printf("%c", a[i]+'A');
		}
		printf("\n%d\n", d);
		return;
	}
	F(i, l) {
		bool ok = true;
		for(int j = d-1; j >= d/2; j--) {
			if (a[j] == i) {
				ok = false;
				F(z, d-j-1) {
					if (a[d-1-z] != a[j-1-z]) {
						ok = true;
						break;
					}
				}
				if (!ok) break;
			}
		}
		if (!ok) continue;
		a[d] = i;
		if (++k <= n) dfs(k, d+1);
	}
}

int main() {
#ifdef LOCAL
    freopen("in", "r", stdin);
    // freopen("out", "w", stdout);
#endif
	while (s(n),s(l) && n) {
		int k = 0;
		dfs(k, 0);
	}
	return 0;
}
